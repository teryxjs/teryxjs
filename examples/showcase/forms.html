Forms & Inputs

Form generation with all field types, validation, and multiple layout modes.

All Field Types

Demonstrates every supported input type in a vertical layout.

View Code 
Teryx.form('#demo-all-fields', {
  action: '/api/form-demo',
  layout: 'vertical',
  submitLabel: 'Submit All',
  fields: [
    { name: 'fullname', label: 'Full Name', type: 'text', placeholder: 'John Doe', required: true },
    { name: 'email', label: 'Email', type: 'email', placeholder: 'john@example.com', required: true },
    { name: 'password', label: 'Password', type: 'password', minLength: 8 },
    { name: 'age', label: 'Age', type: 'number', min: 18, max: 120 },
    { name: 'phone', label: 'Phone', type: 'tel', placeholder: '+1 (555) 000-0000' },
    { name: 'website', label: 'Website', type: 'url', placeholder: 'https://example.com' },
    { name: 'bio', label: 'Bio', type: 'textarea', placeholder: 'Tell us about yourself...', rows: 3 },
    { name: 'role', label: 'Role', type: 'select', options: [
      { label: 'Developer', value: 'dev' },
      { label: 'Designer', value: 'design' },
      { label: 'Manager', value: 'mgr' },
      { label: 'QA Engineer', value: 'qa' },
    ]},
    { name: 'notify', label: 'Email Notifications', type: 'switch' },
    { name: 'agree', label: 'I agree to the terms', type: 'checkbox', required: true },
  ],
});

Form Validation

Client-side validation with required fields, patterns, and length constraints. Try submitting with empty fields.

View Code 
Teryx.form('#demo-validation', {
  action: '/api/form-demo',
  layout: 'vertical',
  submitLabel: 'Register',
  liveValidation: true,
  fields: [
    { name: 'username', label: 'Username', type: 'text', required: true,
      minLength: 3, maxLength: 20, helpText: 'Between 3 and 20 characters' },
    { name: 'email', label: 'Email Address', type: 'email', required: true,
      helpText: 'We will never share your email' },
    { name: 'password', label: 'Password', type: 'password', required: true,
      minLength: 8, helpText: 'Minimum 8 characters' },
    { name: 'confirm', label: 'Confirm Password', type: 'password', required: true },
    { name: 'terms', label: 'I accept the terms and conditions', type: 'checkbox', required: true },
  ],
});

Inline Layout with Field Groups

Compact inline form and grouped fields with collapsible sections.

Field Groups

View Code 
// Inline form
Teryx.form('#demo-inline', {
  action: '/api/form-demo',
  layout: 'inline',
  submitLabel: 'Search',
  fields: [
    { name: 'q', label: 'Keyword', type: 'text', placeholder: 'Search...' },
    { name: 'category', label: 'Category', type: 'select', options: [
      { label: 'All', value: '' },
      { label: 'Users', value: 'users' },
      { label: 'Products', value: 'products' },
    ]},
    { name: 'active', label: 'Active only', type: 'checkbox' },
  ],
});

// Field groups
Teryx.form('#demo-groups', {
  action: '/api/form-demo',
  layout: 'vertical',
  submitLabel: 'Create Account',
  fieldGroups: [
    { name: 'personal', title: 'Personal Information' },
    { name: 'account', title: 'Account Settings' },
    { name: 'prefs', title: 'Preferences', collapsible: true },
  ],
  fields: [
    { name: 'firstName', label: 'First Name', type: 'text', fieldGroup: 'personal', required: true },
    { name: 'lastName', label: 'Last Name', type: 'text', fieldGroup: 'personal', required: true },
    { name: 'dob', label: 'Date of Birth', type: 'date', fieldGroup: 'personal' },
    { name: 'email', label: 'Email', type: 'email', fieldGroup: 'account', required: true },
    { name: 'password', label: 'Password', type: 'password', fieldGroup: 'account', required: true },
    { name: 'role', label: 'Role', type: 'select', fieldGroup: 'account', options: [
      { label: 'User', value: 'user' },
      { label: 'Admin', value: 'admin' },
    ]},
    { name: 'theme', label: 'Theme', type: 'select', fieldGroup: 'prefs', options: [
      { label: 'Light', value: 'light' },
      { label: 'Dark', value: 'dark' },
      { label: 'System', value: 'system' },
    ]},
    { name: 'newsletter', label: 'Subscribe to newsletter', type: 'checkbox', fieldGroup: 'prefs' },
  ],
});
